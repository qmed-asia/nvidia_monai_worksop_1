import logging
from pathlib import Path
from typing import Any, Dict, Optional, Union

import lib.infers
from monai.networks.nets import UNet

from monailabel.interfaces.config import TaskConfig
from monailabel.interfaces.tasks.infer_v2 import InferTask
from monailabel.interfaces.tasks.train import TrainTask
from monailabel.utils.others.generic import strtobool

logger = logging.getLogger(__name__)


class TrainedSpleenSegmentation(TaskConfig):
    """
    TaskConfig that wires a custom spleen UNet checkpoint exported by spleen_segmentation_3d.ipynb.
    """

    def __init__(self):
        super().__init__()
        self.target_spacing = (1.0, 1.0, 1.0)
        self.roi_size = (160, 160, 160)

    def init(self, name: str, model_dir: str, conf: Dict[str, str], planner: Any, **kwargs):
        super().init(name, model_dir, conf, planner, **kwargs)

        self.labels = {"spleen": 1}
        self.network = UNet(
            spatial_dims=3,
            in_channels=1,
            out_channels=2,
            channels=[16, 32, 64, 128, 256],
            strides=[2, 2, 2, 2],
            num_res_units=2,
            norm="batch",
        )

        self.path = self._resolve_paths()
        if not self.path:
            raise FileNotFoundError(
                "No checkpoint found for Trained Spleen segmentation. "
                "Provide --conf trained_spleen_checkpoint=/path/to/best_metric_model.pth."
            )

    def _resolve_paths(self):
        paths = []
        pretrained = Path(self.model_dir) / "pretrained_segmentation_spleen.pt"
        if pretrained.exists():
            paths.append(str(pretrained))

        published = Path(self.model_dir) / f"{self.name}.pt"
        if published.exists():
            paths.append(str(published))

        workspace_ckpt = Path(__file__).resolve().parents[4] / "best_metric_model.pth"
        if workspace_ckpt.exists():
            logger.info("Found notebook checkpoint for spleen at %s", workspace_ckpt)
            paths.append(str(workspace_ckpt))

        custom_ckpt = self._sanitize_path(
            self.conf.get("trained_spleen_checkpoint") or self.conf.get(f"{self.name}_checkpoint")
        )
        if custom_ckpt:
            custom_path = Path(custom_ckpt).expanduser()
            if custom_path.exists():
                paths.append(str(custom_path))
            else:
                logger.warning("Configured trained_spleen checkpoint not found: %s", custom_path)

        return paths

    @staticmethod
    def _sanitize_path(value: Optional[str]) -> Optional[str]:
        if not value:
            return None
        value = value.strip().strip("\"'")
        if len(value) > 2 and value[0] == "r" and ((value[1].isalpha() and value[2] == ":") or value[1] in ("/", "\\")):
            value = value[1:]
        return value

    def infer(self) -> Union[InferTask, Dict[str, InferTask]]:
        task: InferTask = lib.infers.SegmentationSpleen(
            path=self.path,
            network=self.network,
            roi_size=self.roi_size,
            target_spacing=self.target_spacing,
            labels=self.labels,
            preload=strtobool(self.conf.get("preload", "false")),
        )
        return task

    def trainer(self) -> Optional[TrainTask]:
        return None
